-- LocalScript داخل StarterGui

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local TweenService = game:GetService("TweenService")

-- إنشاء GUI

gui.ResetOnSpawn = false

-- كلمة RIP بالأعلى
local ripLabel = Instance.new("TextLabel")

ripLabel.Position = UDim2.new(0.5, -100, 0.02, 0)
ripLabel.BackgroundTransparency = 1
ripLabel.Text = "RIP"
ripLabel.Font = Enum.Font.GothamBlack
ripLabel.TextScaled = true
ripLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
ripLabel.Parent = gui

-- تدرج اللون لكلمة RIP
task.spawn(function()
	while true do
		local toBlack = TweenService:Create(ripLabel, TweenInfo.new(1.5), {TextColor3 = Color3.fromRGB(0, 0, 0)})
		toBlack:Play(); toBlack.Completed:Wait()
		local toWhite = TweenService:Create(ripLabel, TweenInfo.new(1.5), {TextColor3 = Color3.fromRGB(255, 255, 255)})
		toWhite:Play(); toWhite.Completed:Wait()
	end
end)

-- Ping وهمي + FPS وهمي + Timer HH:MM:SS
local fakePing = 70
local fakeFPS = 60
local totalTime = 0

RunService.RenderStepped:Connect(function(delta)
	totalTime += delta

	fakePing = 50 + math.random(0, 100) * math.sin(tick()*0.2)
	fakeFPS = 50 + math.random(0, 80) * math.cos(tick()*0.2)

	if fakePing < 30 then fakePing = 30 elseif fakePing > 160 then fakePing = 160 end
	if fakeFPS < 30 then fakeFPS = 30 elseif fakeFPS > 240 then fakeFPS = 240 end

	pingLabel.Text = string.format("Ping: %d", math.floor(fakePing))
	fpsLabel.Text = string.format("FPS: %d", math.floor(fakeFPS))

	local hours = math.floor(totalTime/3600)
	local minutes = math.floor((totalTime%3600)/60)
	local seconds = math.floor(totalTime%60)
	timerLabel.Text = string.format("الوقت: %02d:%02d:%02d", hours, minutes, seconds)
end)
